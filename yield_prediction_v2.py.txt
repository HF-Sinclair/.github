# Filename: yield_prediction_v2.py

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# Load the dataset
data = pd.read_csv('refinery_data.csv')

# Select features and target variable
X = data[['process_temperature', 'catalyst_amount']]  # Added a new feature 'catalyst_amount'
y = data['yield']

# Split the data into training and test sets with a new random state
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=30)  # Changed random state

# Initialize and train the linear regression model
model = LinearRegression()
model.fit(X_train, y_train)

# Predict and evaluate the model
predictions = model.predict(X_test)
mse = mean_squared_error(y_test, predictions)

print(f"Updated Mean Squared Error: {mse:.2f}")
